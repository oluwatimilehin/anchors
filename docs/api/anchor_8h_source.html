243 template <typename T, typename InputType1, typename InputType2>
244 void Anchor<T, InputType1, InputType2>::compute(int stabilizationNumber) {
245  if (d_recomputeId == stabilizationNumber) {
246  // Don't compute a node more than once in the same cycle
247  return;
248  }
249 
250  T newValue;
251  d_recomputeId = stabilizationNumber;
252 
253  d_hasNeverBeenComputed = false;
254 
255  if (d_numDependencies == 0) {
256  return;
257  }
258 
259  if (d_numDependencies == 1) {
260  InputType1 inputVal = d_firstDependency->get();
261  newValue = d_singleInputUpdater(inputVal);
262  } else {
263  InputType1 inputVal = d_firstDependency->get();
264  InputType2 inputVal2 = d_secondDependency->get();
265 
266  newValue = d_dualInputUpdater(inputVal, inputVal2);
267  }
268 
269  if (newValue != d_value) {
270  d_changeId = stabilizationNumber;
271  d_value = newValue;
272  }
273 }

285 template <typename T, typename InputType1, typename InputType2>
286 bool Anchor<T, InputType1, InputType2>::isStale() const {
287  bool recomputeIdLessThanChildChangeId = false;
288 
289  if (d_numDependencies >= 1) {
290  recomputeIdLessThanChildChangeId =
291  d_recomputeId < d_firstDependency->getChangeId();
292 
293  if (!recomputeIdLessThanChildChangeId && d_numDependencies == 2) {
294  recomputeIdLessThanChildChangeId |=
295  d_recomputeId < d_secondDependency->getChangeId();
296  }
297  }
298 
299  return isNecessary() &
300  (d_hasNeverBeenComputed || recomputeIdLessThanChildChangeId);
301 }
