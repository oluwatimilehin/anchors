127template <typename T>
128void Engine::set(AnchorPtr<T>& anchor, T val) {
129 T oldVal = anchor->get();
130
131 if (oldVal == val) return;
132 d_stabilizationNumber++;
133
134 anchor->setChangeId(d_stabilizationNumber);
135 anchor->set(val);
136
137 if (anchor->isNecessary()) {
138 for (const auto& parent : anchor->getDependants()) {
139 if (parent->isNecessary() && !d_recomputeSet.contains(parent)) {
140 d_recomputeHeap.push(parent);
141 d_recomputeSet.insert(parent);
142 }
143 }
144 }
145}

148void Engine::observe(AnchorPtr<T>& anchor) {

150 return;
